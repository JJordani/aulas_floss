\begin{frame}
	\frametitle{Envolvimento dos usuários no desenvolvimento}
    \begin{block}{A importância de \textit{Feedback}}
    \begin{enumerate}
      \item {\bf Qualquer} projeto pode beneficiar-se\vspace{0.1cm}
      % mesmo que não seja de código aberto/livre
      \item Bugs apontados antes do lançamento\vspace{0.1cm}
      \item Problemas de usabilidade corrigidos\vspace{0.1cm}
      \item Quanto mais usuários, melhor para o projeto\vspace{0.1cm}
    \end{enumerate}
    \end{block}\pause

    \begin{block}{A diferença (o diferencial?) de FLOSS}
    \begin{enumerate}
      \item Usuários \bgreen{cientes sobre o desenvolvimento}\vspace{0.1cm}
      \item \textit{Bug reports} mais informativos\vspace{0.1cm}
      %Como o usuário entende sobre um pouco sobre o sistema ou sobre desenvolvimento, em geral, 
      %os bug reports fornecem informações mais úteis sobre como reproduzir o bug e sobre 
      %potenciais causas para o problema.
      \item O usuário não \tred{\bf precisa} ser um desenvolvedor!
      \begin{itemize}
      \item Mas deve {\bf pensar} como um!
      \end{itemize}
    \end{enumerate}
    \end{block}
\end{frame}

    \begin{block}{Hierarquia baseada em mérito}
    \begin{enumerate}

      \item \textit{Committers} e não-\textit{committers}\vspace{0.1cm}
      \begin{itemize}
         \item Às vezes há vários níveis de \textit{committers}
      \end{itemize}
    \end{enumerate}
    \end{block}\pause

    \begin{block}{``Norma sem sanção é como fogo que não queima''?}

      \item Projetos \textit{open source} têm regras
         \begin{itemize}
         \item Como fazer que sejam seguidas?\vspace{0.1cm}
         \end{itemize}
      \item Ditador {\em benevolente}? 
      \begin{itemize}
      \item Tem que ser, senão, enFORKa ele\vspace{0.1cm}
      \end{itemize}
      \item Reputação é {\bf tudo}
         \begin{itemize}
         \item E as regras costumam ser pragmáticas
         \end{itemize}
    \end{enumerate}
    \end{block}
\end{frame}

\begin{frame}
	\frametitle{Thou Shall not Fork (unless you have to)}
	
    Alguns \textit{forks} de projetos famosos e os motivos:
    
    LibreOffice (OpenOffice)
    % Motivo: medo de que a Oracle fechasse o projeto, impusesse restrições ou simplesmente 
    % o abandonasse. Posteriormente, a Oracle simplesmente entregou o projeto para a Apache, 
    % que o rebatizou de Apache OpenOffice.
    
    XEmacs (Emacs)
    
    FreeBSD (386BSD), NetBSD (386BSD), OpenBSD (NetBSD)
    
    EGCS -- The Enhanced/Experimental GNU Compiler System (GCC)
    % Motivo: devido à grande popularidade do GCC, a FSF tornou-se muito preocupada com a 
    % estabilidade do mesmo, o que significava que a inclusão de novas funcionalidades tornou-se
    % menos prioritária e mais cuidadosa, para evitar prejudicar essa estabilidade. Alguns
    % desenvolvedores, ansiosos por introduzir melhorias no tronco do código do compilador, 
    % várias delas já implementadas ao longo dos anos, resolveram criar o EGCS em 1997, que 
    % inicialmente servia para experimentar com novas funcionalidades ainda não consideradas 
    % maduras o suficiente para entrar no GCC. Dizer que o EGCS é um fork não é inteiramente 
    % preciso. Na verdade, ele é o resultado da integração de diversos forks do GCC. Em 1999 
    % tornou-se o GCC oficial. O GCC até então era desenvolvido em um estilo "Catedral". A 
    % partir daí, passou a empregar a abordagem "Bazar", já usada no desenvolvimento do EGCS.
    
    Apache HTTPd (NCSA HTTPd)
    % Motivo: O desenvolvimento do NCSA HTTPd ia muito lento na segunda metade da década de 90. Um
    % grupo de desenvolvedores pegou o código do servidor e criou um fork, o Apache HTTPd. 
    % Hoje em dia, o Apache não inclui mais código originalmente do NCSA HTTPd.  
    
    WebKit (KHTML)
    
    Drizzle (MySQL)
    % Motivo: A Oracle, com a compra da Sun, tornou-se dona do banco de dados comercial mais usado
    % no mundo e do banco de dados livre mais usado no mundo. Alguns desenvolvedores consideraram
    % essa situação menos que ideal e criaram um fork. 
    
    GNU IceCat (rebranding do Firefox, mas mantido atualizado junto à base de código principal)
    %Mais informações: http://en.wikipedia.org/wiki/List_of_software_forks

\end{frame}
